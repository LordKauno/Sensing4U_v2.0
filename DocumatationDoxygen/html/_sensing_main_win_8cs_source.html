311 public void SortSensor()
312 {
313 var sensorData = new List<(string Label, double Value)>();
314 foreach (DataGridViewRow row in dataGridViewSensor.Rows)
315 {
316 if (row.Cells["SensorValue"].Value != null)
317 {
318 sensorData.Add((row.Cells["SensorLabel"].Value.ToString(), Convert.ToDouble(row.Cells["SensorValue"].Value)));
319 }
320 }
321 for (int i = 0; i < sensorData.Count - 1; i++)
322 {
323 for (int j = 0; j < sensorData.Count - i - 1; j++)
324 {
325 if (sensorData[j].Value > sensorData[j + 1].Value)
326 {
327 var temp = sensorData[j];
328 sensorData[j] = sensorData[j + 1];
329 sensorData[j + 1] = temp;
330 }
331 }
332 }
333 dataGridViewSensor.Rows.Clear();
334 foreach (var data in sensorData)
335 {
336 dataGridViewSensor.Rows.Add(data.Label, data.Value);
337 }
338 sorted = true;
339 MessageBox.Show("Data sorted in ascending order.", "Sort Data");
340 }

584 public int BinarySearch(double targetValue)
585 {
586 var sensorData = new List<(string Label, double Value)>();
587
588 // Sort data if not sorted
589 if (!sorted)
590 SortSensor();
591 double searchValue = targetValue;
592 // Extract grid data
593 foreach (DataGridViewRow row in dataGridViewSensor.Rows)
594 {
595 if (row.Cells["SensorValue"].Value != null)
596 {
597 sensorData.Add((row.Cells["SensorLabel"].Value.ToString(),
598 Convert.ToDouble(row.Cells["SensorValue"].Value)));
599 }
600 }
601 int left = 0;
602 int right = sensorData.Count - 1;
603 int foundIndex = -1;
604 // Binary Search
605 while (left <= right)
606 {
607 int mid = left + (right - left) / 2;
608 if (sensorData[mid].Value == searchValue)
609 {
610 foundIndex = mid;
611 break;
612 }
613
614 if (sensorData[mid].Value < searchValue)
615 left = mid + 1;
616 else
617 right = mid - 1;
618 }
619
620 // No match found
621 if (foundIndex == -1)
622 {
623 MessageBox.Show("Value not found in the data.", "Search Result");
624 return -1;
625 }
626
627 // Highlight ALL rows with this value
628 foreach (DataGridViewRow row in dataGridViewSensor.Rows)
629 {
630 double value = Convert.ToDouble(row.Cells["SensorValue"].Value);
631 if (value == searchValue)
632 row.Cells["SensorValue"].Style.BackColor = Color.Yellow;
633 }
634
635 MessageBox.Show($"Value {searchValue} found (highlighted in yellow).", "Search Result");
636 return foundIndex;
637 }
